    if kirjasuurus == "suur":
        suur = lehekylg_arv * 20
        return suur
    
    elif kirjasuurus == "keskmine":
        keskmine = lehekylg_arv * 30
        return keskmine
    
    elif kirjasuurus == "v채ike":
        vaike = lehekylg_arv * 40
        return vaike
    
    else:
        print("Sisesta korrektne v채채rtus")


def teisenda(sec):
    jaaksec = int(sec % 60)
    minut = int(sec / 60)
    tund = int(minut / 60)
    jaakminut = int(minut % 60)
    
    return f"{tund} tundi, {jaakminut} minutit, {jaaksec} sekundit."
